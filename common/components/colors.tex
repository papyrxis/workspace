\usepackage{xcolor}
\usepackage{xstring}

\makeatletter

\newcommand{\defineColorScheme}[1]{%
  \@namedef{colorscheme@#1}{}%
}

\newcommand{\setColorScheme}[1]{%
  \@ifundefined{colorscheme@#1}{%
    \PackageError{colors}{Color scheme '#1' not defined}{}%
  }{%
    \@nameuse{colorscheme@#1}%
  }%
}

\newcommand{\defineSchemeColor}[3]{%
  \expandafter\gdef\csname scheme@#1@#2\endcsname{#3}%
}

\newcommand{\useSchemeColor}[2]{%
  \@nameuse{scheme@#1@#2}%
}

\makeatother

\newcommand{\applyColorScheme}[1]{%
  \definecolor{primary}{RGB}{\useSchemeColor{#1}{primary}}%
  \definecolor{accent}{RGB}{\useSchemeColor{#1}{accent}}%
  \definecolor{subtle}{RGB}{\useSchemeColor{#1}{subtle}}%
  \definecolor{textprimary}{RGB}{\useSchemeColor{#1}{text}}%
  \definecolor{textsecondary}{RGB}{\useSchemeColor{#1}{textsecondary}}%
  \definecolor{border}{RGB}{\useSchemeColor{#1}{border}}%
  \definecolor{pagebackground}{RGB}{\useSchemeColor{#1}{background}}%
  \definecolor{codebackground}{RGB}{\useSchemeColor{#1}{code}}%
}
