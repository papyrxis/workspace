\makeatletter

\newcommand{\provideCommand}[1]{%
  \@ifundefined{#1}{}{\let#1\relax}%
}

\makeatother

% Complexity notation

\renewcommand{\bigO}{\mathcal{O}}

\providecommand{\bigOmega}
\renewcommand{\bigOmega}{\Omega}

\providecommand{\bigTheta}
\renewcommand{\bigTheta}{\Theta}

\providecommand{\littleo}
\renewcommand{\littleo}{o}

\providecommand{\littleomega}
\renewcommand{\littleomega}{\omega}

% Number sets
\providecommand{\Naturals}
\renewcommand{\Naturals}{\mathbb{N}}

\providecommand{\Integers}
\renewcommand{\Integers}{\mathbb{Z}}

\providecommand{\Rationals}
\renewcommand{\Rationals}{\mathbb{Q}}

\providecommand{\Reals}
\renewcommand{\Reals}{\mathbb{R}}

\providecommand{\Complex}
\renewcommand{\Complex}{\mathbb{C}}

% Probability
\providecommand{\Prob}
\renewcommand{\Prob}{\mathbb{P}}

\providecommand{\Expect}
\renewcommand{\Expect}{\mathbb{E}}

\providecommand{\Var}
\renewcommand{\Var}{\operatorname{Var}}
